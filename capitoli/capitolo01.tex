\chapter{Introduzione ai Sistemi di Basi di Dati}

\section{Concetti di Base}
\begin{definition}[Base di Dati]
    Una collezione di dati correlati che rappresenta un mini-mondo, caratterizzata da una proprietà di persistenza e condivisione.
\end{definition}

\section{L'approccio Database vs Approccio File}
Le differenze fondamentali risiedono in:
\begin{itemize}
    \item \textbf{Auto-descrizione}: Presenza di un catalogo (meta-dati).
    \item \textbf{Astrazione}: Indipendenza tra programmi e dati.
    \item \textbf{Integrità}: Supporto a transazioni e controllo della concorrenza.
\end{itemize}
