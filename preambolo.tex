% --- Ambienti in Stile Matematico ---

% --- Configurazione Titoli (Titlesec) ---
\usepackage[explicit]{titlesec}

% Capitolo: Mostra solo il Titolo, nasconde il numero (ma lo mantiene per l'indice)
\titleformat{\chapter}[block]
  {\normalfont\bfseries\Huge}
  {}
  {0pt}
  {\thispagestyle{empty}#1}

\titlespacing*{\chapter}{0pt}{0pt}{4ex}

% Sezioni e Sottosezioni: Mantengono la numerazione (es. 1.1, 1.2)
\titleformat{\section}
  {\normalfont\bfseries\large}
  {\thesection} 
  {1em}
  {#1}

\titleformat{\subsection}
  {\normalfont\bfseries}
  {\thesubsection}
  {1em}
  {#1}

\titlespacing*{\section}{0pt}{1.5ex plus 1ex minus .2ex}{0.8ex plus .2ex}
